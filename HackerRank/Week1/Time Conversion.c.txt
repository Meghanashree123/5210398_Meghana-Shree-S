#include<stdio.h>

char* Time_Conversion(char* t) {     //t=Time
    static char converted[9];
    int hour;
    hour=(t[0]-'0')*10+(t[1]-'0');
    //handle AM/PM
    if(t[8]=='A'){
        if(hour==12)
           hour=0;
           
    }else{//PM
        if(hour!=12)
           hour+=12;
    }
    sprintf(converted,"%02d:%c%c:%c%c",hour,t[3],t[4],t[6],t[7]);
    return converted;
}

int main() {
    char input[11];
    scanf("%s",input);
    printf("%s\n",Time_Conversion(input));
    return 0;
}
