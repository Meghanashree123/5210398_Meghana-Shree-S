#include <stdio.h>
#include <string.h>

int main() {
    int n, q;
    int i,j;
    scanf("%d", &n);

    char strings[n][100];  // assuming max string length = 100
    for (int i = 0; i < n; i++) {
        scanf("%s", strings[i]);
    }

    scanf("%d", &q);
    char queries[q][101];
    for (i = 0; i < q; i++) {
        scanf("%s", queries[i]);
    }

    // Count matches for each query
    for (i = 0; i < q; i++) {
        int Count = 0;
        for ( j = 0; j < n; j++) {
            if (strcmp(queries[i], strings[j]) == 0) {
                Count++;
            }
        }
        printf("%d\n", Count);
    }

    return 0;
}
