#include <stdio.h>
#include <stdlib.h>
void plusMinus(int array_count, int* arr) {   //function definition
    int p= 0;//p=positive
    int n= 0;//n=negative
    int zero = 0;

    for (int i = 0; i < array_count; i++) {
        if (arr[i] > 0) {
            p++;
        } else if (arr[i] < 0) {
            n++;
        } else {
            zero++;
        }
    }

    // Print results with six decimal places
    printf("%.6f\n", (float)p/ array_count);
    printf("%.6f\n", (float)n / array_count);
    printf("%.6f\n", (float)zero / array_count);
}

int main() {
    int n;
    scanf("%d", &n);

    int *arr = malloc(n * sizeof(int));

    for (int i = 0; i < n; i++) {
        scanf("%d", &arr[i]);
    }

    plusMinus(n, arr); //calling function

    free(arr);
    return 0;
}
